 42    def modify_latent(
 43        self,
 44        latent: torch.FloatTensor,
 45        strength: float,
 46    ) -> torch.FloatTensor:
 47        """
 48        Modify a latent vector by adding noise.
 49
 50        Parameters
 51        ----------
 52        latent: torch.FloatTensor
 53            The input latent vector to modify.
 54        strength: float
 55            The strength of the modification, controlling the amount of noise added.
 56
 57        Returns
 58        -------
 59        torch.FloatTensor
 60            Modified latent vector.
 61        """
 62        noise = self.random_tensor(latent.shape)
 63        new_latent = (1 - strength) * latent + strength * noise
 64        new_latent = (new_latent - new_latent.mean()) / new_latent.std()
 65        return new_latent

 67    def embedding_to_latent(
 68        self,
 69        embedding: torch.FloatTensor,
 70        num_inference_steps: int,
 71        guidance_scale: float,
 72        guidance_rescale: float,
 73        latent: torch.FloatTensor,
 74        return_latent_history: bool = False,
 75    ) -> Union[torch.FloatTensor, List[torch.FloatTensor]]:
 76        """
 77        Generate latent by conditioning on prompt embedding using diffusion.
 78
 79        Parameters
 80        ----------
 81        embedding: torch.FloatTensor
 82            Embedding of text prompt.
 83        num_inference_steps: int
 84            Number of diffusion steps to run.
 85        guidance_scale: float
 86            Guidance scale encourages the model to generate images following the prompt
 87            closely, albeit at the cost of image quality.
 88        guidance_rescale: float
 89            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
 90            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
 91        latent: torch.FloatTensor
 92            Latent to start diffusion from.
 93        return_latent_history: bool
 94            Whether to return latent history. If True, return list of all latents
 95            generated during diffusion steps.
 96
 97        Returns
 98        -------
 99        Union[torch.FloatTensor, List[torch.FloatTensor]]
100            Latent generated by diffusion. If return_latent_history is True, return list of
101            all latents generated during diffusion steps.
102        """
103
104        latent = latent.to(self.device)
105
106        # Set number of inference steps
107        self.scheduler.set_timesteps(num_inference_steps)
108        timesteps = self.scheduler.timesteps
109
110        # Scale the latent noise by the standard deviation required by the scheduler
111        latent = latent * self.scheduler.init_noise_sigma
112        latent_history = [latent]
113
114        # Diffusion inference loop
115        for i, timestep in tqdm(list(enumerate(timesteps))):
116            # Duplicate latent to avoid two forward passes to perform classifier free guidance
117            latent_model_input = torch.cat([latent] * 2)
118            latent_model_input = self.scheduler.scale_model_input(
119                latent_model_input, timestep
120            )
121
122            # Predict noise
123            noise_prediction = self.unet(
124                latent_model_input,
125                timestep,
126                encoder_hidden_states=embedding,
127                return_dict=False,
128            )[0]
129
130            # Perform classifier free guidance
131            noise_prediction = self.classifier_free_guidance(
132                noise_prediction, guidance_scale, guidance_rescale
133            )
134
135            # Update latent
136            latent = self.scheduler.step(
137                noise_prediction, timestep, latent, return_dict=False
138            )[0]
139
140            if return_latent_history:
141                latent_history.append(latent)
142
143        return torch.stack(latent_history) if return_latent_history else latent

145    def interpolate_embedding(
146        self,
147        embedding: torch.FloatTensor,
148        interpolation_steps: Union[int, List[int]],
149        interpolation_type: str,
150    ) -> torch.FloatTensor:
151        """
152        Interpolate based on interpolation type.
153
154        Parameters
155        ----------
156        embedding: torch.FloatTensor
157            Embedding of text prompt.
158        interpolation_steps: Union[int, List[int]]
159            Number of interpolation steps to run.
160        embedding_interpolation_type: str
161            Type of interpolation to run. One of ["lerp", "slerp"].
162
163        Returns
164        -------
165        torch.FloatTensor
166            Interpolated embedding.
167        """
168
169        if interpolation_type == "lerp":
170            interpolation_fn = lerp
171        elif interpolation_type == "slerp":
172            interpolation_fn = slerp
173        else:
174            raise ValueError(
175                f"embedding_interpolation_type must be one of ['lerp', 'slerp'], got {interpolation_type}."
176            )
177
178        # Split embedding into unconditional and text embeddings
179        unconditional_embedding, text_embedding = embedding.chunk(2)
180        steps = (
181            torch.linspace(0, 1, interpolation_steps, dtype=embedding.dtype)
182            .cpu()
183            .numpy()
184        )
185        steps = np.expand_dims(steps, axis=tuple(range(1, text_embedding.ndim)))
186        interpolations = []
187
188        # Interpolate between text embeddings
189        # TODO: Think of a better way of doing this
190        # See if it can be done parallelly instead
191        for i in range(text_embedding.shape[0] - 1):
192            interpolations.append(
193                interpolation_fn(
194                    text_embedding[i], text_embedding[i + 1], steps
195                ).squeeze(dim=1)
196            )
197        interpolations = torch.cat(interpolations)
198
199        # TODO: Think of a better way of doing this
200        # It can be done because all unconditional embeddings are the same
201        single_unconditional_embedding = unconditional_embedding[0].unsqueeze(dim=0)
202        unconditional_embedding = single_unconditional_embedding.repeat(
203            interpolations.shape[0], 1, 1
204        )
205        interpolations = torch.cat([unconditional_embedding, interpolations])
206
207        return interpolations

337        prompt: List[str],
338        latent: Optional[torch.FloatTensor] = None,
339        image_height: Optional[int] = None,
340        image_width: Optional[int] = None,
341        num_inference_steps: int = 50,
342        interpolation_steps: Union[int, List[int]] = 8,
343        guidance_scale: float = 7.5,
344        guidance_rescale: float = 0.7,
345        negative_prompt: Optional[List[str]] = None,
346        output_type: str = "pil",
347        return_latent_history: bool = False,
348        embedding_interpolation_type: str = "lerp",
349        latent_interpolation_type: str = "slerp",
350    ) -> OutputType:
351        """
352        Run inference by conditioning on text prompts and interpolating between them.
353
354        Parameters
355        ----------
356        prompt: List[str]
357            List of text prompts to condition on.
358        latent: Optional[torch.FloatTensor]
359            Latents to interpolate between. If None, latents are generated from noise
360            but image_height and image_width must be provided.
361        image_height: Optional[int]
362            Height of image to generate.
363        image_width: Optional[int]
364            Width of image to generate.
365        num_inference_steps: int
366            Number of diffusion steps to run.
367        interpolation_steps: Union[int, List[int]]
368            Number of interpolation steps to run.
369        guidance_scale: float
370            Guidance scale encourages the model to generate images following the prompt
371            closely, albeit at the cost of image quality.
372        guidance_rescale: float
373            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
374            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
375        negative_prompt: Optional[List[str]]
376            Negative text prompts to uncondition on.
377        output_type: str
378            Type of output to return. One of ["latent", "pil", "pt", "np"].
379        return_latent_history: bool
380            Whether to return the latent history. If True, return list of all latents
381            generated during diffusion steps.
382        embedding_interpolation_type: str
383            Type of interpolation to run for text embeddings. One of ["lerp", "slerp"].
384        latent_interpolation_type: str
385            Type of interpolation to run for latents. One of ["lerp", "slerp"].
386
387        Returns
388        -------
389        OutputType
390            Generated output based on output_type.
391        """
392
393        # Validate input
394        self.validate_input(
395            prompt=prompt,
396            negative_prompt=negative_prompt,
397            image_height=image_height,
398            image_width=image_width,
399        )
400
401        # There should be atleast 2 prompts to run interpolation
402        if not isinstance(prompt, list):
403            raise ValueError(f"prompt must be a list of strings, not {type(prompt)}")
404        if len(prompt) < 2:
405            raise ValueError(
406                f"prompt must be a list of at least 2 strings, not {len(prompt)}"
407            )
408        if isinstance(interpolation_steps, int):
409            pass
410            # interpolation_steps = [interpolation_steps] * (len(prompt) - 1)
411        elif isinstance(interpolation_steps, list):
412            if len(interpolation_steps) != len(prompt) - 1:
413                raise ValueError(
414                    f"interpolation_steps must be a list of length len(prompt) - 1, not {len(interpolation_steps)}"
415                )
416            raise NotImplementedError(
417                "interpolation_steps as a list is not yet implemented"
418            )
419        else:
420            raise ValueError(
421                f"interpolation_steps must be an int or list, not {type(interpolation_steps)}"
422            )
423
424        if latent is None:
425            shape = (
426                len(prompt),
427                self.unet.config.in_channels,
428                image_height // self.vae_scale_factor,
429                image_width // self.vae_scale_factor,
430            )
431            latent = self.random_tensor(shape)
432        elif len(prompt) != latent.shape[0]:
433            raise ValueError(
434                f"prompt and latent must be of the same length, not {len(prompt)} and {latent.shape[0]}"
435            )
436
437        # Generate embedding to condition on prompt and uncondition on negative prompt
438        embedding = self.prompt_to_embedding(
439            prompt=prompt,
440            negative_prompt=negative_prompt,
441        )
442
443        # Interpolate between embeddings
444        interpolated_embedding = self.interpolate_embedding(
445            embedding=embedding,
446            interpolation_steps=interpolation_steps,
447            interpolation_type=embedding_interpolation_type,
448        )
449
450        # Interpolate between latents
451        interpolated_latent = self.interpolate_latent(
452            latent=latent,
453            interpolation_steps=interpolation_steps,
454            interpolation_type=latent_interpolation_type,
455        )
456
457        # Run inference
458        latent = self.embedding_to_latent(
459            embedding=interpolated_embedding,
460            num_inference_steps=num_inference_steps,
461            guidance_scale=guidance_scale,
462            guidance_rescale=guidance_rescale,
463            latent=interpolated_latent,
464            return_latent_history=return_latent_history,
465        )
466
467        return self.resolve_output(
468            latent=latent,
469            output_type=output_type,
470            return_latent_history=return_latent_history,
471        )

 78        Generate latent by conditioning on prompt embedding using diffusion.
 79
 80        Parameters
 81        ----------
 82        embedding: torch.FloatTensor
 83            Embedding of text prompt.
 84        num_inference_steps: int
 85            Number of diffusion steps to run.
 86        guidance_scale: float
 87            Guidance scale encourages the model to generate images following the prompt
 88            closely, albeit at the cost of image quality.
 89        guidance_rescale: float
 90            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
 91            Flawed](https://arxiv.org/pdf/2305.08891.pdf).

 95            Whether to return latent history. If True, return list of all latents

 99        -------
100        Union[torch.FloatTensor, List[torch.FloatTensor]]

102            all latents generated during diffusion steps.
103        """
104
105        latent = latent.to(self.device)
106
107        # Set number of inference steps
108        self.scheduler.set_timesteps(num_inference_steps)
109        timesteps = self.scheduler.timesteps
110
111        # Scale the latent noise by the standard deviation required by the scheduler
112        latent = latent * self.scheduler.init_noise_sigma
113        latent_history = [latent]
114
115        # Diffusion inference loop
116        for i, timestep in tqdm(list(enumerate(timesteps))):
117            # Duplicate latent to avoid two forward passes to perform classifier free guidance
118            latent_model_input = torch.cat([latent] * 2)
119            latent_model_input = self.scheduler.scale_model_input(
120                latent_model_input, timestep
121            )
122
123            # Predict noise
124            noise_prediction = self.unet(
125                latent_model_input,
126                timestep,
127                encoder_hidden_states=embedding,
128                return_dict=False,
129            )[0]
130
131            # Perform classifier free guidance
132            noise_prediction = self.classifier_free_guidance(
133                noise_prediction, guidance_scale, guidance_rescale
134            )
135
136            # Update latent
137            latent = self.scheduler.step(
138                noise_prediction, timestep, latent, return_dict=False
139            )[0]
140
141            if return_latent_history:
142                latent_history.append(latent)
143
144        return torch.stack(latent_history) if return_latent_history else latent

375            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
376        negative_prompt: Optional[List[str]]
377            Negative text prompts to uncondition on.
378        output_type: str
379            Type of output to return. One of ["latent", "pil", "pt", "np"].
380        return_latent_history: bool
381            Whether to return the latent history. If True, return list of all latents
382            generated during diffusion steps.
383        embedding_interpolation_type: str
384            Type of interpolation to run for text embeddings. One of ["lerp", "slerp"].
385        latent_interpolation_type: str
386            Type of interpolation to run for latents. One of ["lerp", "slerp"].
387
388        Returns
389        -------
390        OutputType
391            Generated output based on output_type.
392        """

394        # Validate input
395        self.validate_input(
396            prompt=prompt,
397            negative_prompt=negative_prompt,
398            image_height=image_height,
399            image_width=image_width,
400        )
401
402        # There should be atleast 2 prompts to run interpolation
403        if not isinstance(prompt, list):
404            raise ValueError(f"prompt must be a list of strings, not {type(prompt)}")
405        if len(prompt) < 2:
406            raise ValueError(
407                f"prompt must be a list of at least 2 strings, not {len(prompt)}"
408            )
409        if isinstance(interpolation_steps, int):
410            pass
411            # interpolation_steps = [interpolation_steps] * (len(prompt) - 1)
412        elif isinstance(interpolation_steps, list):
413            if len(interpolation_steps) != len(prompt) - 1:
414                raise ValueError(
415                    f"interpolation_steps must be a list of length len(prompt) - 1, not {len(interpolation_steps)}"

420        else:

423            )
424
425        if latent is None:
426            shape = (
427                len(prompt),
428                self.unet.config.in_channels,
429                image_height // self.vae_scale_factor,
430                image_width // self.vae_scale_factor,
431            )
432            latent = self.random_tensor(shape)
433        elif len(prompt) != latent.shape[0]:
434            raise ValueError(
435                f"prompt and latent must be of the same length, not {len(prompt)} and {latent.shape[0]}"
436            )
437
438        # Generate embedding to condition on prompt and uncondition on negative prompt
439        embedding = self.prompt_to_embedding(
440            prompt=prompt,
441            negative_prompt=negative_prompt,
442        )
443
444        # Interpolate between embeddings
445        interpolated_embedding = self.interpolate_embedding(
446            embedding=embedding,
447            interpolation_steps=interpolation_steps,
448            interpolation_type=embedding_interpolation_type,
449        )

453            latent=latent,
454            interpolation_steps=interpolation_steps,
455            interpolation_type=latent_interpolation_type,
456        )
457
458        # Run inference
459        latent = self.embedding_to_latent(
460            embedding=interpolated_embedding,
461            num_inference_steps=num_inference_steps,
462            guidance_scale=guidance_scale,
463            guidance_rescale=guidance_rescale,
464            latent=interpolated_latent,
465            return_latent_history=return_latent_history,
466        )
467

469            latent=latent,
470            output_type=output_type,
471            return_latent_history=return_latent_history,
472        )
