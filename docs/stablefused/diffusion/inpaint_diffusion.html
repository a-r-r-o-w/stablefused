 49    def embedding_to_latent(
 50        self,
 51        embedding: torch.FloatTensor,
 52        num_inference_steps: int,
 53        start_step: int,
 54        guidance_scale: float,
 55        guidance_rescale: float,
 56        latent: torch.FloatTensor,
 57        mask: torch.FloatTensor,
 58        masked_image_latent: torch.FloatTensor,
 59        return_latent_history: bool = False,
 60    ) -> Union[torch.FloatTensor, List[torch.FloatTensor]]:
 61        """
 62        Generate latent by conditioning on prompt embedding and input latent using diffusion.
 63
 64        Parameters
 65        ----------
 66        embedding: torch.FloatTensor
 67            Embedding of text prompt.
 68        num_inference_steps: int
 69            Number of diffusion steps to run.
 70        start_step: int
 71            Step to start diffusion from. The higher the value, the more similar the generated
 72            image will be to the input image.
 73        guidance_scale: float
 74            Guidance scale encourages the model to generate images following the prompt
 75            closely, albeit at the cost of image quality.
 76        guidance_rescale: float
 77            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
 78            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
 79        latent: torch.FloatTensor
 80            Latent to start diffusion from.
 81        mask: torch.FloatTensor
 82            Mask to condition on. Values below 0.5 are treated as 0 and values above 0.5
 83            are treated as 1. Values that are 1 (white) will be inpainted and values that
 84            are 0 (black) will be preserved.
 85        masked_image_latent: torch.FloatTensor
 86            Latent of masked image.
 87        return_latent_history: bool
 88            Whether to return latent history. If True, return list of all latents
 89            generated during diffusion steps.
 90
 91        Returns
 92        -------
 93        Union[torch.FloatTensor, List[torch.FloatTensor]]
 94            Latent generated by diffusion. If return_latent_history is True, return list of
 95            all latents generated during diffusion steps.
 96        """
 97
 98        # Set number of inference steps
 99        self.scheduler.set_timesteps(num_inference_steps)
100
101        # Add noise to latent based on start step
102        start_timestep = (
103            self.scheduler.timesteps[start_step].repeat(latent.shape[0]).long()
104        )
105        noise = self.random_tensor(latent.shape)
106        if start_step > 0:
107            latent = self.scheduler.add_noise(latent, noise, start_timestep)
108        else:
109            latent = noise * self.scheduler.init_noise_sigma
110
111        timesteps = self.scheduler.timesteps[start_step:]
112        latent_history = [latent]
113
114        # Concatenate mask and masked_image_latent as required by classifier free guidance
115        mask = torch.cat([mask] * 2)
116        masked_image_latent = torch.cat([masked_image_latent] * 2)
117
118        # Diffusion inference loop
119        for i, timestep in tqdm(list(enumerate(timesteps))):
120            # Duplicate latent to avoid two forward passes to perform classifier free guidance
121            latent_model_input = torch.cat([latent] * 2)
122            latent_model_input = self.scheduler.scale_model_input(
123                latent_model_input, timestep
124            )
125
126            # TODO: The image inpainting model "runwayml/stable-diffusion-inpainting" requires
127            # 9 channels as input. The first 4 channels are the image latent, the next 1 channel
128            # is the mask and the last 4 channels are the masked image latent. This assumes that
129            # all models have the same number of input channels but that may not be the case.
130            # Refactor this logic to be more generic.
131            latent_model_input = torch.cat(
132                [latent_model_input, mask, masked_image_latent], dim=1
133            )
134
135            # Predict noise
136            noise_prediction = self.unet(
137                latent_model_input,
138                timestep,
139                encoder_hidden_states=embedding,
140                return_dict=False,
141            )[0]
142
143            # Perform classifier free guidance
144            noise_prediction = self.classifier_free_guidance(
145                noise_prediction, guidance_scale, guidance_rescale
146            )
147
148            # Update latent
149            latent = self.scheduler.step(
150                noise_prediction, timestep, latent, return_dict=False
151            )[0]
152
153            if return_latent_history:
154                latent_history.append(latent)
155
156        return torch.stack(latent_history) if return_latent_history else latent

158    def _handle_preprocess_tensor(
159        self,
160        image: torch.Tensor,
161        mask: torch.Tensor,
162    ) -> Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]:
163        """
164        Helper function to preprocess tensors representing images. Note that
165        this should only be used for images already preprocessed with relevant logic in
166        the _handle_preprocess_pil and _handle_preprocess_numpy functions.
167        """
168
169        if image.ndim != 4 or mask.ndim != 4:
170            raise ValueError(
171                f"image and mask must have 4 dimensions, got {image.ndim} and {mask.ndim}"
172            )
173        if image.shape[0] != mask.shape[0]:
174            raise ValueError(
175                f"image and mask must have same batch size, got {image.shape[0]} and {mask.shape[0]}"
176            )
177        if image.shape[1] != 3 or mask.shape[1] != 1:
178            raise ValueError(
179                f"image must have 3 channels and mask must have 1 channel, got {image.shape[1]} and {mask.shape[1]}"
180            )
181        if image.shape[2] != mask.shape[2] or image.shape[3] != mask.shape[3]:
182            raise ValueError(
183                f"image and mask must have same height and width, got {image.shape[2:]} and {mask.shape[2:]}"
184            )
185        if image.min() < 0 or image.max() > 255:
186            raise ValueError(
187                f"image must have values between 0 and 255, got {image.min()} and {image.max()}"
188            )
189        if mask.min() < 0 or mask.max() > 255:
190            raise ValueError(
191                f"mask must have values between 0 and 255, got {mask.min()} and {mask.max()}"
192            )
193
194        image = (image / 255.0) * 2 - 1
195        mask = mask / 255.0
196
197        mask[mask < 0.5] = 0
198        mask[mask >= 0.5] = 1
199
200        image.to(device=self.device, dtype=self.torch_dtype)
201        mask.to(device=self.device, dtype=self.torch_dtype)
202
203        masked_image = image * (1 - mask)
204
205        return image, mask, masked_image

207    def _handle_preprocess_numpy(
208        self,
209        image: np.ndarray,
210        mask: np.ndarray,
211    ) -> Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]:
212        """
213        Helper function to preprocess numpy arrays representing images. Note that
214        this should only be used for images already preprocessed with relevant logic in
215        the _handle_preprocess_pil function.
216        """
217
218        if image.ndim == 3:
219            image = np.expand_dims(image, axis=0)
220        if mask.ndim == 3:
221            mask = np.expand_dims(mask, axis=0)
222        if image.ndim != 4 or mask.ndim != 4:
223            raise ValueError(
224                f"image and mask must have 4 dimensions, got {image.ndim} and {mask.ndim}"
225            )
226
227        image = image.transpose(0, 3, 1, 2)
228        image = torch.from_numpy(image).to(device=self.device, dtype=self.torch_dtype)
229
230        mask = mask.transpose(0, 3, 1, 2)
231        mask = torch.from_numpy(mask).to(device=self.device, dtype=self.torch_dtype)
232
233        return self._handle_preprocess_tensor(image, mask)

259        translation: int,
260        translation_frames: int,
261    ) -> List[int]:
262        """Helper function to calculate translation per frame."""
263        step_size = translation // translation_frames
264        remainder = translation % translation_frames

268    @staticmethod
269    def _translate_image_and_mask(
270        image: Image.Image,
271        walk_type: InpaintWalkType,
272        translation: Union[int, Tuple[int, int]],
273        mask: Optional[np.ndarray] = None,
274    ) -> Tuple[Image.Image, Image.Image]:
275        """Helper function to translate image and mask in given direction by specified translation."""
276
277        def apply_translation(dx: int, dy: int) -> Image.Image:
278            return image.transform(
279                (image.width, image.height),
280                Image.AFFINE,
281                (1, 0, dx, 0, 1, dy),
282                resample=Image.BICUBIC,
283            )
284
285        if walk_type == InpaintWalkType.UP:
286            if mask is not None:
287                mask[:translation, :] = 255
288            new_image = apply_translation(0, -translation)
289
290        elif walk_type == InpaintWalkType.DOWN:
291            if mask is not None:
292                mask[-translation:, :] = 255
293            new_image = apply_translation(0, translation)
294
295        elif walk_type == InpaintWalkType.LEFT:
296            if mask is not None:
297                mask[:, :translation] = 255
298            new_image = apply_translation(-translation, 0)
299
300        elif walk_type == InpaintWalkType.RIGHT:
301            if mask is not None:
302                mask[:, -translation:] = 255
303            new_image = apply_translation(translation, 0)
304
305        elif (
306            walk_type == InpaintWalkType.FORWARD
307            or walk_type == InpaintWalkType.BACKWARD
308        ):
309            tw, th = translation
310
311            if mask is not None:
312                mask[:th, :] = 255
313                mask[-th:, :] = 255
314                mask[:, :tw] = 255
315                mask[:, -tw:] = 255
316            downsampled_image = image.resize(
317                (image.width - 2 * tw, image.height - 2 * th), resample=Image.LANCZOS
318            )
319            new_image = Image.new("RGB", (image.width, image.height))
320            new_image.paste(downsampled_image, (tw, th))
321
322        return new_image, Image.fromarray(mask) if mask is not None else None

324    @staticmethod
325    def _generate_filler_frames(
326        start_image: Image.Image,
327        end_image: Image.Image,
328        walk_type: InpaintWalkType,
329        actual_translation: Union[int, Tuple[int, int]],
330        filler_translations: Union[int, List[int]],
331    ) -> List[Image.Image]:
332        """Helper function to generate filler frames for given walk type."""
333
334        if (
335            walk_type == InpaintWalkType.FORWARD
336            or walk_type == InpaintWalkType.BACKWARD
337        ):
338            if not isinstance(filler_translations, int):
339                raise ValueError(
340                    f"filler_translations must be of type int for InpaintWalkType.FORWARD or InpaintWalkType.BACKWARD, got {type(filler_translations)}"
341                )
342        else:
343            if not isinstance(filler_translations, list):
344                raise ValueError(
345                    f"filler_translations must be of type list for InpaintWalkType.UP, InpaintWalkType.DOWN, InpaintWalkType.LEFT or InpaintWalkType.RIGHT, got {type(filler_translations)}"
346                )
347
348        frames = []
349        width = start_image.width
350        height = start_image.height
351
352        if walk_type == InpaintWalkType.UP:
353            for filler_translation in filler_translations:
354                a = start_image.crop((0, 0, width, height - filler_translation))
355                b = end_image.crop(
356                    (
357                        0,
358                        actual_translation - filler_translation,
359                        width,
360                        actual_translation,
361                    )
362                )
363                result_img = Image.new("RGB", (width, height))
364                result_img.paste(b, (0, 0))
365                result_img.paste(a, (0, b.height))
366                frames.append(result_img)
367
368        elif walk_type == InpaintWalkType.DOWN:
369            for filler_translation in filler_translations:
370                a = start_image.crop((0, filler_translation, width, height))
371                b = end_image.crop(
372                    (
373                        0,
374                        height - actual_translation,
375                        width,
376                        height - actual_translation + filler_translation,
377                    )
378                )
379                result_img = Image.new("RGB", (width, height))
380                result_img.paste(a, (0, 0))
381                result_img.paste(b, (0, a.height))
382                frames.append(result_img)
383
384        elif walk_type == InpaintWalkType.LEFT:
385            for filler_translation in filler_translations:
386                a = start_image.crop((0, 0, width - filler_translation, height))
387                b = end_image.crop(
388                    (
389                        actual_translation - filler_translation,
390                        0,
391                        actual_translation,
392                        height,
393                    )
394                )
395                result_img = Image.new("RGB", (width, height))
396                result_img.paste(b, (0, 0))
397                result_img.paste(a, (b.width, 0))
398                frames.append(result_img)
399
400        elif walk_type == InpaintWalkType.RIGHT:
401            for filler_translation in filler_translations:
402                a = start_image.crop((filler_translation, 0, width, height))
403                b = end_image.crop(
404                    (
405                        width - actual_translation,
406                        0,
407                        width - actual_translation + filler_translation,
408                        height,
409                    )
410                )
411                result_img = Image.new("RGB", (width, height))
412                result_img.paste(a, (0, 0))
413                result_img.paste(b, (a.width, 0))
414                frames.append(result_img)
415
416        elif (
417            walk_type == InpaintWalkType.FORWARD
418            or walk_type == InpaintWalkType.BACKWARD
419        ):
420            aw, ah = actual_translation
421            width_factor = 1 - 2 * aw / width
422            height_factor = 1 - 2 * ah / height
423            width_crop_factor = width - 2 * aw
424            height_crop_factor = height - 2 * ah
425
426            translated_image, _ = InpaintDiffusion._translate_image_and_mask(
427                start_image, walk_type, actual_translation, mask=None
428            )
429            translated_image = translated_image.convert("RGBA")
430            translated_image = np.array(translated_image)
431            translated_image[:ah, :, 3] = 0
432            translated_image[-ah:, :, 3] = 0
433            translated_image[:, :aw, 3] = 0
434            translated_image[:, -aw:, 3] = 0
435            translated_image = Image.fromarray(translated_image)
436
437            end_image.paste(translated_image, mask=translated_image)
438
439            for i in range(filler_translations - 1):
440                translation_factor = 1 - (i + 1) / filler_translations
441                interpolation_image = end_image
442                interpolation_width = round(
443                    (1 - width_factor**translation_factor) * width / 2
444                )
445                interpolation_height = round(
446                    (1 - height_factor**translation_factor) * height / 2
447                )
448
449                interpolation_image = interpolation_image.crop(
450                    (
451                        interpolation_width,
452                        interpolation_height,
453                        width - interpolation_width,
454                        height - interpolation_height,
455                    )
456                ).resize((width, height), resample=Image.LANCZOS)
457
458                w = width - 2 * interpolation_width
459                h = height - 2 * interpolation_height
460                crop_fix_width = round((1 - width_crop_factor / w) * width / 2)
461                crop_fix_height = round((1 - height_crop_factor / h) * height / 2)
462
463                start_image_crop_fix, _ = InpaintDiffusion._translate_image_and_mask(
464                    start_image, walk_type, (crop_fix_width, crop_fix_height), mask=None
465                )
466                start_image_crop_fix = start_image_crop_fix.convert("RGBA")
467                start_image_crop_fix = np.array(start_image_crop_fix)
468                start_image_crop_fix[:crop_fix_height, :, 3] = 0
469                start_image_crop_fix[-crop_fix_height:, :, 3] = 0
470                start_image_crop_fix[:, :crop_fix_width, 3] = 0
471                start_image_crop_fix[:, -crop_fix_width:, 3] = 0
472                start_image_crop_fix = Image.fromarray(start_image_crop_fix)
473
474                interpolation_image.paste(
475                    start_image_crop_fix, mask=start_image_crop_fix
476                )
477                frames.append(interpolation_image)
478
479            frames.append(end_image)
480
481        return frames

483    def preprocess_image_and_mask(
484        self,
485        image: ImageType,
486        mask: ImageType,
487        image_height: int,
488        image_width: int,
489    ) -> Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]:
490        """
491        Preprocess image and mask to be used for inference.
492
493        Parameters
494        ----------
495        image: ImageType
496            Input image to condition on.
497        mask: ImageType
498            Input mask to condition on. Must have same height and width as image.
499            Must have 1 channel. Must have values between 0 and 1. Values below 0.5
500            are treated as 0 and values above 0.5 are treated as 1. Values that are
501            1 (white) will be inpainted and values that are 0 (black) will be
502            preserved.
503        image_height: int
504            Height of image to generate. If height of provided image and image_height
505            do not match, image will be resized to image_height using PIL Lanczos method.
506        image_width: int
507            Width of image to generate. If width of provided image and image_width
508            do not match, image will be resized to image_width using PIL Lanczos method.
509
510        Returns
511        -------
512        Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]
513            Tuple of image, mask and masked_image.
514        """
515        if image is None or mask is None:
516            raise ValueError("image and mask cannot be None")
517        if type(image) != type(mask):
518            raise TypeError(
519                f"image and mask must be of same type, got {type(image)} and {type(mask)}"
520            )
521        if isinstance(image, Image.Image):
522            image = [image]
523            mask = [mask]
524
525        if isinstance(image, torch.Tensor):
526            image, mask, masked_image = self._handle_preprocess_tensor(image, mask)
527        elif isinstance(image, np.ndarray):
528            image, mask, masked_image = self._handle_preprocess_numpy(image, mask)
529        else:
530            image, mask, masked_image = self._handle_preprocess_pil(
531                image, mask, image_height, image_width
532            )
533
534        return image, mask, masked_image

536    @torch.no_grad()
537    def __call__(
538        self,
539        prompt: PromptType,
540        image: ImageType,
541        mask: ImageType,
542        image_height: int = 512,
543        image_width: int = 512,
544        num_inference_steps: int = 50,
545        start_step: int = 0,
546        guidance_scale: float = 7.5,
547        guidance_rescale: float = 0.7,
548        negative_prompt: Optional[PromptType] = None,
549        output_type: str = "pil",
550        return_latent_history: bool = False,
551    ) -> OutputType:
552        """
553        Run inference by conditioning on input image, mask and prompt.
554
555        Parameters
556        ----------
557        prompt: PromptType
558            Text prompt to condition on.
559        image: ImageType
560            Input image to condition on.
561        mask: ImageType
562            Input mask to condition on. Must have same height and width as image.
563            Must have 1 channel. Must have values between 0 and 1. Values below 0.5
564            are treated as 0 and values above 0.5 are treated as 1. Values that are
565            1 (white) will be inpainted and values that are 0 (black) will be
566            preserved.
567        image_height: int
568            Height of image to generate. If height of provided image and image_height
569            do not match, image will be resized to image_height using PIL Lanczos method.
570        image_width: int
571            Width of image to generate. If width of provided image and image_width
572            do not match, image will be resized to image_width using PIL Lanczos method.
573        num_inference_steps: int
574            Number of diffusion steps to run.
575        start_step: int
576            Step to start diffusion from. The higher the value, the more similar the generated
577            image will be to the input image.
578        guidance_scale: float
579            Guidance scale encourages the model to generate images following the prompt
580            closely, albeit at the cost of image quality.
581        guidance_rescale: float
582            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
583            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
584        negative_prompt: Optional[PromptType]
585            Negative text prompt to uncondition on.
586        output_type: str
587            Type of output to return. One of ["latent", "pil", "pt", "np"].
588        return_latent_history: bool
589            Whether to return the latent history. If True, return list of all latents
590            generated during diffusion steps.
591
592        Returns
593        -------
594        OutputType
595            Generated output based on output_type.
596        """
597
598        # Validate input
599        self.validate_input(
600            prompt=prompt,
601            negative_prompt=negative_prompt,
602            image_height=image_height,
603            image_width=image_width,
604            start_step=start_step,
605            num_inference_steps=num_inference_steps,
606        )
607
608        # Preprocess image and mask
609        image, mask, masked_image = self.preprocess_image_and_mask(
610            image=image,
611            mask=mask,
612            image_height=image_height,
613            image_width=image_width,
614        )
615
616        # Generate latent from input image and masked_image. Mask is down/up-scaled to match latent shape
617        image_latent = self.image_to_latent(image)
618        mask = F.interpolate(mask, size=(image_latent.shape[2], image_latent.shape[3]))
619        masked_image_latent = self.image_to_latent(masked_image)
620
621        # Generate embedding to condition on prompt and uncondition on negative prompt
622        embedding = self.prompt_to_embedding(
623            prompt=prompt,
624            negative_prompt=negative_prompt,
625        )
626
627        # Run inference
628        latent = self.embedding_to_latent(
629            embedding=embedding,
630            num_inference_steps=num_inference_steps,
631            start_step=start_step,
632            guidance_scale=guidance_scale,
633            guidance_rescale=guidance_rescale,
634            latent=image_latent,
635            mask=mask,
636            masked_image_latent=masked_image_latent,
637            return_latent_history=return_latent_history,
638        )
639
640        return self.resolve_output(
641            latent=latent,
642            output_type=output_type,
643            return_latent_history=return_latent_history,
644        )

684        height_translation_per_step: int
685            Number of pixels to translate image up/down per step.
686        width_translation_per_step: int
687            Number of pixels to translate image left/right per step.
688        translation_factor: Optional[float]
689            Factor to translate image by. If provided, overrides height_translation_per_step
690            and width_translation_per_step. Must be between 0 and 1.
691        num_inpainting_steps: int
692            Number of inpainting steps to run.
693        interpolation_steps: int
694            Number of interpolation steps to run between inpainting steps.
695        num_inference_steps: int

697        start_step: int

703        guidance_rescale: float

752                    f"prompt must have length of num_inpainting_steps, got {len(walk_type)} and {num_inpainting_steps}"
753                )
754
755        if negative_prompt is not None:
756            if isinstance(negative_prompt, str):
757                negative_prompt = [negative_prompt]
758            if len(negative_prompt) == 1:
759                negative_prompt = negative_prompt * num_inpainting_steps
760            if len(negative_prompt) != num_inpainting_steps:
761                raise ValueError(
762                    f"negative_prompt must have length of num_inpainting_steps, got {len(walk_type)} and {num_inpainting_steps}"
763                )
764

766        walk_has_forward = InpaintWalkType.FORWARD in walk_type

775                )

779                    f"height_translation_per_step must be less than image_height, got {height_translation_per_step} and {image_height}"
780                )
781            if width_translation_per_step >= image_width:

783                    f"width_translation_per_step must be less than image_width, got {width_translation_per_step} and {image_width}"
784                )
785

787            translation=height_translation_per_step,
788            translation_frames=interpolation_steps,
789        )
790        width_filler_translations = self._calculate_translation_per_frame(

792            translation_frames=interpolation_steps,
793        )
794

797            width_filler_translations[i] += width_filler_translations[i - 1]
798
799        assert height_filler_translations[-1] == height_translation_per_step
800        assert width_filler_translations[-1] == width_translation_per_step
801
802        image = image.resize(
803            (image_height, image_width), resample=Image.LANCZOS
804        ).convert("RGB")
805
806        base_mask = np.zeros((image_width, image_height), dtype=np.uint8)
807        prev_image = image
808        frames = []
809
810        for prompt, negative_prompt, walk in tqdm(zip(prompt, negative_prompt, walk_type)):
811            if walk == InpaintWalkType.LEFT or walk == InpaintWalkType.RIGHT:
812                translation = width_translation_per_step
813                filler_translations = width_filler_translations
814            elif walk == InpaintWalkType.UP or walk == InpaintWalkType.DOWN:
815                translation = height_translation_per_step
816                filler_translations = height_filler_translations
817            elif walk == InpaintWalkType.BACKWARD:
818                translation = (width_translation_per_step, height_translation_per_step)
819                filler_translations = interpolation_steps
820            elif walk == InpaintWalkType.FORWARD:
821                raise ValueError(
822                    "InpaintWalkType.FORWARD is not supported yet. If you would like to do this, reverse the sequence of images generated using InpaintWalkType.BACKWARD"
823                )
824            else:
825                raise ValueError(f"Invalid Inpaint Walk Type: {walk}")
826
827            image, mask = self._translate_image_and_mask(
828                prev_image, walk, translation, mask=base_mask.copy()
829            )
830            generated_image = self.generate(
831                prompt=prompt,
832                image=image,
833                mask=mask,
834                image_height=image_height,
835                image_width=image_width,
836                num_inference_steps=num_inference_steps,
837                start_step=start_step,
838                guidance_scale=guidance_scale,
839                guidance_rescale=guidance_rescale,
840                negative_prompt=negative_prompt,
841                output_type="pil",
842                return_latent_history=False,
843            )[0]
844
845            filler_frames_list = self._generate_filler_frames(
846                start_image=prev_image,
847                end_image=generated_image,
848                walk_type=walk,
849                actual_translation=translation,
850                filler_translations=filler_translations,
851            )
852
853            prev_image = filler_frames_list[-1].copy()
854
855            if walk == InpaintWalkType.FORWARD:
856                filler_frames_list = filler_frames_list[::-1]
857
858            frames.extend(filler_frames_list)
859
860        return frames

 50    def embedding_to_latent(
 51        self,
 52        embedding: torch.FloatTensor,
 53        num_inference_steps: int,
 54        start_step: int,
 55        guidance_scale: float,
 56        guidance_rescale: float,
 57        latent: torch.FloatTensor,
 58        mask: torch.FloatTensor,
 59        masked_image_latent: torch.FloatTensor,
 60        return_latent_history: bool = False,
 61    ) -> Union[torch.FloatTensor, List[torch.FloatTensor]]:
 62        """
 63        Generate latent by conditioning on prompt embedding and input latent using diffusion.
 64
 65        Parameters
 66        ----------
 67        embedding: torch.FloatTensor
 68            Embedding of text prompt.
 69        num_inference_steps: int
 70            Number of diffusion steps to run.
 71        start_step: int
 72            Step to start diffusion from. The higher the value, the more similar the generated
 73            image will be to the input image.
 74        guidance_scale: float
 75            Guidance scale encourages the model to generate images following the prompt
 76            closely, albeit at the cost of image quality.
 77        guidance_rescale: float
 78            Guidance rescale from [Common Diffusion Noise Schedules and Sample Steps are
 79            Flawed](https://arxiv.org/pdf/2305.08891.pdf).
 80        latent: torch.FloatTensor
 81            Latent to start diffusion from.
 82        mask: torch.FloatTensor
 83            Mask to condition on. Values below 0.5 are treated as 0 and values above 0.5
 84            are treated as 1. Values that are 1 (white) will be inpainted and values that
 85            are 0 (black) will be preserved.
 86        masked_image_latent: torch.FloatTensor
 87            Latent of masked image.
 88        return_latent_history: bool
 89            Whether to return latent history. If True, return list of all latents
 90            generated during diffusion steps.
 91
 92        Returns
 93        -------
 94        Union[torch.FloatTensor, List[torch.FloatTensor]]
 95            Latent generated by diffusion. If return_latent_history is True, return list of
 96            all latents generated during diffusion steps.
 97        """
 98
 99        # Set number of inference steps
100        self.scheduler.set_timesteps(num_inference_steps)
101
102        # Add noise to latent based on start step
103        start_timestep = (
104            self.scheduler.timesteps[start_step].repeat(latent.shape[0]).long()
105        )
106        noise = self.random_tensor(latent.shape)
107        if start_step > 0:
108            latent = self.scheduler.add_noise(latent, noise, start_timestep)
109        else:
110            latent = noise * self.scheduler.init_noise_sigma
111
112        timesteps = self.scheduler.timesteps[start_step:]
113        latent_history = [latent]
114
115        # Concatenate mask and masked_image_latent as required by classifier free guidance
116        mask = torch.cat([mask] * 2)
117        masked_image_latent = torch.cat([masked_image_latent] * 2)
118
119        # Diffusion inference loop
120        for i, timestep in tqdm(list(enumerate(timesteps))):
121            # Duplicate latent to avoid two forward passes to perform classifier free guidance
122            latent_model_input = torch.cat([latent] * 2)
123            latent_model_input = self.scheduler.scale_model_input(
124                latent_model_input, timestep
125            )
126
127            # TODO: The image inpainting model "runwayml/stable-diffusion-inpainting" requires
128            # 9 channels as input. The first 4 channels are the image latent, the next 1 channel
129            # is the mask and the last 4 channels are the masked image latent. This assumes that
130            # all models have the same number of input channels but that may not be the case.
131            # Refactor this logic to be more generic.
132            latent_model_input = torch.cat(
133                [latent_model_input, mask, masked_image_latent], dim=1
134            )
135
136            # Predict noise
137            noise_prediction = self.unet(
138                latent_model_input,
139                timestep,
140                encoder_hidden_states=embedding,
141                return_dict=False,
142            )[0]
143
144            # Perform classifier free guidance
145            noise_prediction = self.classifier_free_guidance(
146                noise_prediction, guidance_scale, guidance_rescale
147            )
148
149            # Update latent
150            latent = self.scheduler.step(
151                noise_prediction, timestep, latent, return_dict=False
152            )[0]
153
154            if return_latent_history:
155                latent_history.append(latent)
156
157        return torch.stack(latent_history) if return_latent_history else latent

159    def _handle_preprocess_tensor(
160        self,
161        image: torch.Tensor,
162        mask: torch.Tensor,
163    ) -> Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]:
164        """
165        Helper function to preprocess tensors representing images. Note that
166        this should only be used for images already preprocessed with relevant logic in
167        the _handle_preprocess_pil and _handle_preprocess_numpy functions.
168        """
169
170        if image.ndim != 4 or mask.ndim != 4:
171            raise ValueError(
172                f"image and mask must have 4 dimensions, got {image.ndim} and {mask.ndim}"
173            )
174        if image.shape[0] != mask.shape[0]:
175            raise ValueError(
176                f"image and mask must have same batch size, got {image.shape[0]} and {mask.shape[0]}"
177            )
178        if image.shape[1] != 3 or mask.shape[1] != 1:
179            raise ValueError(
180                f"image must have 3 channels and mask must have 1 channel, got {image.shape[1]} and {mask.shape[1]}"
181            )
182        if image.shape[2] != mask.shape[2] or image.shape[3] != mask.shape[3]:
183            raise ValueError(
184                f"image and mask must have same height and width, got {image.shape[2:]} and {mask.shape[2:]}"
185            )
186        if image.min() < 0 or image.max() > 255:
187            raise ValueError(
188                f"image must have values between 0 and 255, got {image.min()} and {image.max()}"
189            )
190        if mask.min() < 0 or mask.max() > 255:
191            raise ValueError(
192                f"mask must have values between 0 and 255, got {mask.min()} and {mask.max()}"
193            )
194
195        image = (image / 255.0) * 2 - 1
196        mask = mask / 255.0
197
198        mask[mask < 0.5] = 0
199        mask[mask >= 0.5] = 1
200
201        image.to(device=self.device, dtype=self.torch_dtype)
202        mask.to(device=self.device, dtype=self.torch_dtype)
203
204        masked_image = image * (1 - mask)
205
206        return image, mask, masked_image

269    @staticmethod
270    def _translate_image_and_mask(
271        image: Image.Image,
272        walk_type: InpaintWalkType,
273        translation: Union[int, Tuple[int, int]],
274        mask: Optional[np.ndarray] = None,
275    ) -> Tuple[Image.Image, Image.Image]:
276        """Helper function to translate image and mask in given direction by specified translation."""
277
278        def apply_translation(dx: int, dy: int) -> Image.Image:
279            return image.transform(
280                (image.width, image.height),
281                Image.AFFINE,
282                (1, 0, dx, 0, 1, dy),
283                resample=Image.BICUBIC,
284            )
285
286        if walk_type == InpaintWalkType.UP:
287            if mask is not None:
288                mask[:translation, :] = 255
289            new_image = apply_translation(0, -translation)
290
291        elif walk_type == InpaintWalkType.DOWN:
292            if mask is not None:
293                mask[-translation:, :] = 255
294            new_image = apply_translation(0, translation)
295
296        elif walk_type == InpaintWalkType.LEFT:
297            if mask is not None:
298                mask[:, :translation] = 255
299            new_image = apply_translation(-translation, 0)
300
301        elif walk_type == InpaintWalkType.RIGHT:
302            if mask is not None:
303                mask[:, -translation:] = 255
304            new_image = apply_translation(translation, 0)
305
306        elif (
307            walk_type == InpaintWalkType.FORWARD
308            or walk_type == InpaintWalkType.BACKWARD
309        ):
310            tw, th = translation
311
312            if mask is not None:
313                mask[:th, :] = 255
314                mask[-th:, :] = 255
315                mask[:, :tw] = 255
316                mask[:, -tw:] = 255
317            downsampled_image = image.resize(
318                (image.width - 2 * tw, image.height - 2 * th), resample=Image.LANCZOS
319            )
320            new_image = Image.new("RGB", (image.width, image.height))
321            new_image.paste(downsampled_image, (tw, th))
322
323        return new_image, Image.fromarray(mask) if mask is not None else None

326    def _generate_filler_frames(
327        start_image: Image.Image,
328        end_image: Image.Image,
329        walk_type: InpaintWalkType,
330        actual_translation: Union[int, Tuple[int, int]],
331        filler_translations: Union[int, List[int]],
332    ) -> List[Image.Image]:
333        """Helper function to generate filler frames for given walk type."""
334

336            walk_type == InpaintWalkType.FORWARD
337            or walk_type == InpaintWalkType.BACKWARD
338        ):
339            if not isinstance(filler_translations, int):
340                raise ValueError(
341                    f"filler_translations must be of type int for InpaintWalkType.FORWARD or InpaintWalkType.BACKWARD, got {type(filler_translations)}"
342                )
343        else:
344            if not isinstance(filler_translations, list):
345                raise ValueError(
346                    f"filler_translations must be of type list for InpaintWalkType.UP, InpaintWalkType.DOWN, InpaintWalkType.LEFT or InpaintWalkType.RIGHT, got {type(filler_translations)}"
347                )
348
349        frames = []
350        width = start_image.width
351        height = start_image.height
352
353        if walk_type == InpaintWalkType.UP:
354            for filler_translation in filler_translations:
355                a = start_image.crop((0, 0, width, height - filler_translation))
356                b = end_image.crop(
357                    (
358                        0,
359                        actual_translation - filler_translation,
360                        width,
361                        actual_translation,
362                    )
363                )
364                result_img = Image.new("RGB", (width, height))
365                result_img.paste(b, (0, 0))
366                result_img.paste(a, (0, b.height))
367                frames.append(result_img)
368
369        elif walk_type == InpaintWalkType.DOWN:
370            for filler_translation in filler_translations:
371                a = start_image.crop((0, filler_translation, width, height))
372                b = end_image.crop(
373                    (
374                        0,
375                        height - actual_translation,
376                        width,
377                        height - actual_translation + filler_translation,
378                    )
379                )
380                result_img = Image.new("RGB", (width, height))
381                result_img.paste(a, (0, 0))
382                result_img.paste(b, (0, a.height))
383                frames.append(result_img)
384
385        elif walk_type == InpaintWalkType.LEFT:
386            for filler_translation in filler_translations:
387                a = start_image.crop((0, 0, width - filler_translation, height))
388                b = end_image.crop(
389                    (
390                        actual_translation - filler_translation,
391                        0,
392                        actual_translation,
393                        height,
394                    )
395                )
396                result_img = Image.new("RGB", (width, height))
397                result_img.paste(b, (0, 0))
398                result_img.paste(a, (b.width, 0))
399                frames.append(result_img)
400
401        elif walk_type == InpaintWalkType.RIGHT:
402            for filler_translation in filler_translations:
403                a = start_image.crop((filler_translation, 0, width, height))
404                b = end_image.crop(
405                    (
406                        width - actual_translation,
407                        0,
408                        width - actual_translation + filler_translation,
409                        height,
410                    )
411                )
412                result_img = Image.new("RGB", (width, height))
413                result_img.paste(a, (0, 0))
414                result_img.paste(b, (a.width, 0))
415                frames.append(result_img)
416
417        elif (
418            walk_type == InpaintWalkType.FORWARD
419            or walk_type == InpaintWalkType.BACKWARD
420        ):
421            aw, ah = actual_translation
422            width_factor = 1 - 2 * aw / width
423            height_factor = 1 - 2 * ah / height
424            width_crop_factor = width - 2 * aw
425            height_crop_factor = height - 2 * ah
426
427            translated_image, _ = InpaintDiffusion._translate_image_and_mask(
428                start_image, walk_type, actual_translation, mask=None
429            )
430            translated_image = translated_image.convert("RGBA")
431            translated_image = np.array(translated_image)
432            translated_image[:ah, :, 3] = 0
433            translated_image[-ah:, :, 3] = 0
434            translated_image[:, :aw, 3] = 0
435            translated_image[:, -aw:, 3] = 0
436            translated_image = Image.fromarray(translated_image)
437
438            end_image.paste(translated_image, mask=translated_image)
439
440            for i in range(filler_translations - 1):
441                translation_factor = 1 - (i + 1) / filler_translations
442                interpolation_image = end_image
443                interpolation_width = round(
444                    (1 - width_factor**translation_factor) * width / 2
445                )
446                interpolation_height = round(
447                    (1 - height_factor**translation_factor) * height / 2
448                )
449

452                        interpolation_width,
453                        interpolation_height,

455                        height - interpolation_height,
456                    )
457                ).resize((width, height), resample=Image.LANCZOS)

460                h = height - 2 * interpolation_height
461                crop_fix_width = round((1 - width_crop_factor / w) * width / 2)
462                crop_fix_height = round((1 - height_crop_factor / h) * height / 2)

464                start_image_crop_fix, _ = InpaintDiffusion._translate_image_and_mask(
465                    start_image, walk_type, (crop_fix_width, crop_fix_height), mask=None
466                )
467                start_image_crop_fix = start_image_crop_fix.convert("RGBA")
468                start_image_crop_fix = np.array(start_image_crop_fix)
469                start_image_crop_fix[:crop_fix_height, :, 3] = 0
470                start_image_crop_fix[-crop_fix_height:, :, 3] = 0
471                start_image_crop_fix[:, :crop_fix_width, 3] = 0
472                start_image_crop_fix[:, -crop_fix_width:, 3] = 0
473                start_image_crop_fix = Image.fromarray(start_image_crop_fix)

475                interpolation_image.paste(
476                    start_image_crop_fix, mask=start_image_crop_fix
477                )
478                frames.append(interpolation_image)
479
480            frames.append(end_image)
